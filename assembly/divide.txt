        lw      0    3    n        # reg(1) = n (dividend)
        lw      0    2    d        # reg(2) = d (divisor)
        lw      0    1    quo     # reg(3) = 0 (quotient)
        lw      0    4    neg1     # reg(4) = -1 (used for subtraction)
        lw      0    5    pos1     # reg(5) = 1 (used for incrementing quotient)
        lw      0    6    negd    # reg(6) = -d (used for incrementing quotient)
        lw      0    7    temp
loop    beq     3    2    done    # If n < d, exit loop
        add     3    6    3        # n = n - d (simulate n -= d with -1 increments)
        add     1    5    1        # quotient += 1
        beq     0    0    loop     # Repeat loop
done    halt                       # End program
n       .fill   15                 # Example dividend: 15
d       .fill   3                  # Example divisor: 3
neg1    .fill   -1                 # Constant -1 for subtraction
quo    .fill   0                  # Constant 0 to initialize quotient
pos1 .fill 1
negd .fill -3
temp .fill 0
